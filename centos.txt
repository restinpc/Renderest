Очистка диска
yum install yum-utils
find /var -name "*.log" \( \( -size +10M -mtime +7 \) -o -mtime +30 \) -exec truncate {} --size 0 \;
yum clean all
rm -rf /var/cache/yum
rm -rf /var/tmp/yum-*
find -regex ".*/core.[0-9]+$" -delete
rm -rf /root/.npm /home/*/.npm /root/.node-gyp /home/*/.node-gyp /tmp/npm-*
rm -rf /home/*/.cache/*/* /root/.cache/*/*
